       ```python
       import asyncio
       import logging
       import threading
       from contextlib import asynccontextmanager
       from fastapi import FastAPI

       from pnpg.config import load_config
       from pnpg.prereqs import check_npcap, check_admin
       from pnpg.capture.interface import select_interface
       from pnpg.capture.sniffer import sniffer_supervisor
       from pnpg.pipeline.worker import pipeline_worker

       logging.basicConfig(
           level=logging.DEBUG,
           format="%(asctime)s [%(levelname)s] %(name)s: %(message)s",
       )

       @asynccontextmanager
       async def lifespan(app: FastAPI):
           # Startup sequence: config -> prereqs -> interface -> sniffer -> pipeline
           config = load_config()                    # CONFIG-01/02
           check_npcap()                             # CAP-02
           check_admin()                             # CAP-03
           iface = select_interface(config)          # CAP-04/09

           loop = asyncio.get_running_loop()
           queue = asyncio.Queue(maxsize=config["queue_size"])  # CAP-05
           drop_counter = [0]
           stop_event = threading.Event()

           supervisor_task = asyncio.create_task(
               sniffer_supervisor(loop, queue, iface, config, drop_counter, stop_event)
           )
           worker_task = asyncio.create_task(
               pipeline_worker(queue, config)
           )

           app.state.queue = queue
           app.state.config = config
           app.state.drop_counter = drop_counter
           app.state.stop_event = stop_event

           yield  # Application runs

           # Shutdown
           stop_event.set()
           supervisor_task.cancel()
           worker_task.cancel()
           try:
               await supervisor_task
           except asyncio.CancelledError:
               pass
           try:
               await worker_task
           except asyncio.CancelledError:
               pass

       app = FastAPI(title="PNPG", lifespan=lifespan)

       @app.get("/status")
       async def status():
           return {"status": "running"}
       ```
