    Imports: asyncio, logging, time, psutil

    logger = logging.getLogger(__name__)

    VALID_STATUSES = frozenset({"ESTABLISHED", "SYN_SENT", "CLOSE_WAIT"})

    Function 1 — _refresh_cache(cache: dict, config: dict) -> None:
    - ttl_secs = config.get("proc_cache_ttl_sec", 2)
    - expires_at = time.monotonic() + ttl_secs
    - Wrap psutil.net_connections(kind="inet") in try/except psutil.AccessDenied — on error, log warning and RETURN (do NOT clear cache) per PROC-04
    - Build new_cache = {} by iterating connections:
      - Skip if not conn.laddr (no local address)
      - Skip if conn.status not in VALID_STATUSES (filters out 0.0.0.0 laddr mismatch per Research Pitfall 2)
      - key = (conn.laddr.ip, conn.laddr.port)
      - If conn.pid is None: store {"pid": -1, "process_name": "unknown process", "expires_at": expires_at}
      - Else: try psutil.Process(conn.pid).name() — on NoSuchProcess or AccessDenied, set name="unknown process", pid=-1 per PROC-04
      - Store {"pid": conn_pid, "process_name": name, "expires_at": expires_at}
    - Atomic swap: cache.clear(); cache.update(new_cache) (NEVER reassign cache reference — see Research Pitfall 5)

    Function 2 — async def process_poller_loop(cache: dict, config: dict) -> None:
    - interval = config.get("poll_interval_ms", 200) / 1000.0
    - while True loop:
      - try: _refresh_cache(cache, config); await asyncio.sleep(interval)
      - except asyncio.CancelledError: break (clean shutdown)
      - except Exception: logger.critical("process_poller_loop error", exc_info=True); await asyncio.sleep(interval)

    Function 3 — def enrich_event(event: dict, cache: dict) -> dict:
    - key = (event.get("src_ip"), event.get("src_port"))
    - entry = cache.get(key)
    - If entry is not None and time.monotonic() < entry["expires_at"]:
      return {**event, "process_name": entry["process_name"], "pid": entry["pid"]}
    - Else (cache miss or expired — PROC-03/PROC-06):
      return {**event, "process_name": "unknown process", "pid": -1}

    CRITICAL immutability: enrich_event returns a NEW dict via {**event, ...}. Never mutate event.
